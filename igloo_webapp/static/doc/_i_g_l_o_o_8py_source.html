168  """
169  This function updates the position of the animal. This is important, as
170  the ends of the gradient are reflective, e.g.: The gradient is 50 mm
171  long the fly is at 47 mm and makes a 10 mm step towards the near end.
172  In this case the animal walk 3 mm to the near end and gets reflected
173  for 7 mm. The new fly position would be 43 mm.
174 
175  The result is saved in self.position

183  if (newPos > gradientDist):
184  overshoot = newPos - gradientDist
185  newPos = gradientDist-overshoot
186  elif (newPos < 0.0):

202  def drosoTbyConduction(self):
203  """
204  This function calculates the temperature change for animals in our TLM
205  model. It uses Newtons law for convection and models the Drosophila as an
206  3 by 0.5 mm cylinder walking in a 3 mm wide cylinder. Internally the
207  values given will be transferred to the correct dimensions. The
208  following values are set conductance:
209  air 0.0262 W/(m²*K) David R. Lide (Hrsg.): CRC Handbook of Chemistry and Physics. 90. Auflage. (Internet-Version: 2010), CRC Press/Taylor and Francis, Boca Raton, FL, Fluid Properties, S. 6-184. Werte gelten bei 300 K.
210  water 0.6 W/(m²*K) https://de.wikipedia.org/wiki/Eigenschaften_des_Wassers#W.C3.A4rmeleitf.C3.A4higkeit
211 
212  The formula for rate of heat flow is: dQ T1-T2
213  Q = -- = lambda * A* -----
214  dt D
215 
216  where D is the wallt to wall thickness of the object lambda is its
217  conductance A its surface T1, T2 the temperature of the object and its
218  surroundings.
219 
220  Drosophila surface' as a cylinder would be for the cylinders hull:
221  2*pi*r*l and the two disks: (2*pi*r**2)*2. If the cylinder has an
222  r = 0.5mm and a length of l =2mm the resulting surface is 7.85 mm²
223 
224  To calculate the temperature change we have to know the conductance of
225  our fly which we approximate as a cylinder of water lambda 0.6. A
226  cylinder of that size would have a mass of 1.57 mg
227 
228  Model limits: Drosophila is a 1 times 2 mm cylinder consisting of
229  1.58 mg of water suspended in air.
230 
231  The result is saved in self.drosoT
232  """
233  #physical size of the conductors
234  conductance = 0.6 #np.array([0.37, 0.0262])
235  surface = 7.85
236 
237  #time spent in the new environment
238  if (len(self.tempTrace.shape) >1): # for all after the first iteration
239  t = self.time -self.tempTrace[-2,0]
240  else: # first iteration
241  t = self.time
242  # convert to correct dimension
243 
244  # The formula for rate of heat flow is: dQ T1-T2
245  # Q = -- = lambda * A* -----
246  # d D
247  # where D is the wallt to wall thickness of the object lambda is its
248  # conductance A its surface T1, T2 the temperature of the object and its
249  # surroundings.
250  # C to K because the Celsius and Kalvin have the same relative units
251 
252  dT= self.ambientT - self.drosoT
253  # surface mm² to m²
254  surface = surface / 10.0**6.0
255  # mm to m
256  D = 1.0 / 10.0**3.0
257 
258 
259  # Calculate the heat flow
260  Q = conductance*surface*(dT/D)*t
261 
262  # Q has the dimension Watt*secvParamsond = Joule
263 
264  # Joule = energy needed to change 1 gr of waters temperature by 0.239 K
265  # Drosophila weighs approx. 0.25mg and consists mainly out of water and
266  # therefore 1 Joule should heat up a 1.57 mg mass of water by 152.23 °K
267  temperature_change = Q*152.23
268 
269  # if our temperature change overshoots it will actually be limited to
270  # the ambient temperature
271  if (self.ambientT > self.drosoT) and (self.drosoT+temperature_change> self.ambientT):
272  self.drosoT = self.ambientT
273  elif (self.ambientT < self.drosoT) and (self.drosoT+temperature_change < self.ambientT):
274  self.drosoT = self.ambientT
275  else:
276  self.drosoT = self.drosoT + temperature_change

281  def simulateSingleFly(self):
282  """
283  This function simulates the random walk of a single fly. For the
284  duration given by self.walkDur in [s] on the gradient defined
285  by self.gradientExt and self.gradientDist
286  To simulate more than one fly, please use simulateFlyPopulation
287 
288  The result is saved in self.tempTrace
289 
290  self.tempTrace is a nx4 vector, where column 1) time [s], 2) position
291  [mm], 3) ambient temperature [°C], 4) body temperature [°C] and n is
292  self.walkDur*self.sps
293  """
294  # reset single trace variable
295  self.resetTempTrace()
296  #calculate rearing temperature dependent function paramteres
297  self.vParams,self.dParams = lI.calcParameters(self.rearingT)
298  # simulate random walk until the duration is exceeded
299  while (self.time < self.walkDur):
300  # make a step
301  self.stepFunc()
302  # update fly position and ambient temperature
303  self.updatePosition()
304  self.updateAmbientTemp()
305  # change Drosophilas temperature by conduction
306  self.drosoTbyConduction()
307  # add sample to vector
308  self.tempTrace = np.vstack((self.tempTrace,np.array([self.time,self.position,self.ambientT, self.drosoT])))
309 
310 
311  #interpolate the data to fix samplerate
312  xi = np.linspace(0,self.walkDur,self.walkDur*self.sps) # time vector
313  pos = np.interp(xi,self.tempTrace[:,0],self.tempTrace[:,1]) # positions
314  aT = np.interp(xi,self.tempTrace[:,0],self.tempTrace[:,2]) # ambient Temperature
315  dT = np.interp(xi,self.tempTrace[:,0],self.tempTrace[:,3]) # animal Temperature
316 
317  #built return matrix
318  self.tempTrace = np.column_stack((xi,pos))
319  self.tempTrace = np.column_stack((self.tempTrace,aT))
320  self.tempTrace = np.column_stack((self.tempTrace,dT))
